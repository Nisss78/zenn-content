# 【実践OpenClaw】日常業務を自動化する7つの実用的なワークフロー

こんにちは、OpenClawを活用して業務効率化を目指す開発者さんへ。

この記事では、OpenClawを使った**日常業務の自動化**に焦点を当てます。単なるAPI連携やファイル操作ではなく、実際の業務で役立つ具体的なワークフローを7つ紹介します。初心者の方でもすぐに実践できるように、実用的なコード例と共に解説します。

## はじめに：なぜ日常業務の自動化なのか

多くの開発者やデータサイエンティストが、以下のような繰り返し業務に悩んでいませんか？

- 毎日のレポート生成
- データの整形と変換
- ファイルの整理とバックアップ
- メールや通知の確認
- コードレビューの準備

これらの作業は1つ1つは簡単でも、毎日繰り返すと膨大な時間を消費します。OpenClawを使えば、これらの作業を**自動化**し、本来の創造的な作業に集中できます。

## ワークフロー1：毎日の進捗レポート自動生成

### ケーススタディ
プロジェクトの進捗を毎日レポートする必要がある場合、手作業でまとめるのは大変です。OpenClawを使えば、各種ファイルから情報を自動収集してレポートを生成できます。

### 実装コード

```python
import re
from datetime import datetime, timedelta

    progress_info = collect_project_progress(project_dir)

{progress_info['in_progress_tasks']}

## 3. GitHub Issuesの進捗
{task_progress}

## 4. 次日の予定
{progress_info['next_tasks']}

## 5. 障害要因
{progress_info['blockers']}

    report_file = f"{output_dir}/daily_report_{today}.md"
    write(file_path=report_file, content=report_content)
    
    return report_file

def collect_project_progress(project_dir):
    """プロジェクトファイルから進捗情報を収集"""
    progress_info = {
        'completed_tasks': "",
        'in_progress_tasks': "",
        'next_tasks': "",
        'blockers': ""
    }
    
    # README.mdから進捗情報を抽出
    readme_path = f"{project_dir}/README.md"
    if os.path.exists(readme_path):
        content = read(file_path=readme_path)
        progress_info = extract_progress_from_readme(content)
    
    return progress_info

def extract_progress_from_readme(content):
    """READMEから進捗情報を抽出"""
    # 簡単な正規表現で進捗を抽出
    completed = re.findall(r'- \[x\] (.+)', content)
    in_progress = re.findall(r'- \[\~\] (.+)', content)
    next_tasks = re.findall(r'- \[ \] (.+)', content)
    
    return {
        'completed_tasks': "\n".join(f"- {task}" for task in completed),
        'in_progress_tasks': "\n".join(f"- {task}" for task in in_progress),
        'next_tasks': "\n".join(f"- {task}" for task in next_tasks),
        'blockers': "特になし"
    }

# 使用例
report_file = generate_daily_report("my-project")
print(f"レポート生成完了: {report_file}")
```

- **簡単なカスタマイズ**: 抽出ルールを自由に変更可能

## ワークフロー2：データ変換と整形バッチ処理

### ケーススタディ
複数のCSVファイルを整形して分析用のデータセットを作成する必要がある場合。特に異なるフォーマットのファイルを統一する作業は時間がかかります。

### 実装コード

```python
import csv
import json
from pathlib import Path

def batch_data_conversion(input_dir, output_dir="converted_data", target_format="json"):
    """
    複数のデータファイルを一括変換する
    
    Args:
        input_dir (str): 入力ディレクトリ
        output_dir (str): 出力ディレクトリ
        target_format (str): 変換先フォーマット
    """
    # 出力ディレクトリを作成
    os.makedirs(output_dir, exist_ok=True)
    
    # 入力ディレクトリ内のファイルを探索
    input_files = list(Path(input_dir).glob("*.csv")) + list(Path(input_dir).glob("*.json"))
    
    conversion_count = 0
    
    for input_file in input_files:
        try:
            # ファイルを変換
            converted_data = convert_data_file(input_file, target_format)
            
            # 出力ファイル名を生成
            output_filename = f"{input_file.stem}_converted.{target_format}"
            output_file = Path(output_dir) / output_filename
            
            # 変換データを保存
            if target_format == "json":
                write(file_path=str(output_file), content=json.dumps(converted_data, indent=2, ensure_ascii=False))
            else:
                write(file_path=str(output_file), content=converted_data)
            
            conversion_count += 1
            print(f"変換完了: {input_file.name} -> {output_filename}")
            
        except Exception as e:
            print(f"変換失敗: {input_file.name} - {e}")
    
    print(f"合計 {conversion_count} ファイルを変換しました")
    return conversion_count

def convert_data_file(input_file, target_format):
    """単一のデータファイルを変換する"""
    input_path = str(input_file)
    
    if input_file.suffix == '.csv':
        # CSVファイルをJSONに変換
        data = read_csv_file(input_path)
        if target_format == "json":
            return data
        else:
            return convert_to_other_format(data, target_format)
    
    elif input_file.suffix == '.json':
        # JSONファイルをCSVに変換
        data = json.loads(read(file_path=input_path))
        if target_format == "csv":
            return convert_to_csv(data)
        else:
            return convert_to_other_format(data, target_format)
    
    else:
        raise ValueError(f"サポートされていないファイル形式: {input_file.suffix}")

def read_csv_file(file_path):
    """CSVファイルを読み込む"""
    content = read(file_path=file_path)
    csv_reader = csv.reader(content.split('\n'))
    headers = next(csv_reader)
    
    data = []
    for row in csv_reader:
        if len(row) == len(headers):
            data.append(dict(zip(headers, row)))
    
    return data

def convert_to_csv(data):
    """データをCSV形式に変換する"""
    if not data:
        return ""
    
    # ヘッダーを抽出
    headers = list(data[0].keys())
    
    # CSV形式の文字列を生成
    csv_lines = [','.join(headers)]
    
    for row in data:
        csv_line = ','.join(str(row.get(header, '')) for header in headers)
        csv_lines.append(csv_line)
    
    return '\n'.join(csv_lines)

# 使用例

    input_dir="raw_data",
    output_dir="processed_data",
    target_format="json"
)
```

- **エラーハンドリング**: 失敗したファイルでも処理を続行

## ワークフロー3：ファイル整理と自動バックアップ

### ケーススタディ
プロジェクトファイルが増えてくると、どのファイルがどの目的で使われているか分からなくなります。OpenClawを使えば、ファイルを自動的に整理・分類できます。

### 実装コード

```python
import shutil
from datetime import datetime
import hashlib

def organize_project_files(source_dir, structure_config=None):
    """
    プロジェクトファイルを自動整理する
    
    Args:
        source_dir (str): 整理対象ディレクトリ
        structure_config (dict): 整理構成の設定
    """
    # デフォルトの整理構成
    if structure_config is None:
        structure_config = {
            'documents': ['.md', '.txt', '.doc', '.pdf'],
            'images': ['.png', '.jpg', '.jpeg', '.gif', '.svg'],
            'code': ['.py', '.js', '.html', '.css', '.json'],
            'data': ['.csv', '.xlsx', '.json', '.xml'],
            'config': ['.yaml', '.yml', '.env', '.conf'],
            'temp': ['.tmp', '.log', '.bak']
        }
    
    # 整理先のディレクトリを作成
    organized_dir = f"{source_dir}_organized"
    os.makedirs(organized_dir, exist_ok=True)
    
    # 各カテゴリ用のディレクトリを作成
    for category in structure_config.keys():
        category_dir = os.path.join(organized_dir, category)
        os.makedirs(category_dir, exist_ok=True)
    
    # ファイルを整理
    moved_files = 0
    
    for file_path in Path(source_dir).rglob("*"):
        if file_path.is_file():
            category = determine_file_category(file_path, structure_config)
            if category:
                # ファイルを移動
                dest_path = Path(organized_dir) / category / file_path.name
                shutil.move(str(file_path), str(dest_path))
                moved_files += 1
                print(f"移動: {file_path.name} -> {category}/")
    
    # 整理レポートを生成
    generate_organize_report(organized_dir, moved_files)
    
    return organized_dir

def determine_file_category(file_path, structure_config):
    """ファイルの拡張子からカテゴリを判定"""
    extension = file_path.suffix.lower()
    
    for category, extensions in structure_config.items():
        if extension in extensions:
            return category
    
    return None

def generate_organize_report(organized_dir, moved_files):
    """整理レポートを生成"""
    report_content = f"""# ファイル整理レポート

## 整理概要
- 整理日時: {datetime.now().strftime("%Y-%m-%d %H:%M:%S")}
- 移動ファイル数: {moved_files}
- 整理先ディレクトリ: {organized_dir}

## 整理結果

"""
    
    # 各カテゴリのファイル数をカウント
    for category in ['documents', 'images', 'code', 'data', 'config', 'temp']:
        category_dir = Path(organized_dir) / category
        if category_dir.exists():
            file_count = len(list(category_dir.glob("*")))
            report_content += f"- {category}: {file_count} ファイル\n"
    
    # レポートを保存
    report_file = Path(organized_dir) / "整理レポート.md"
    write(file_path=str(report_file), content=report_content)

def create_backup(source_dir, backup_dir="backups"):
    """ディレクトリのバックアップを作成"""
    timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
    backup_name = f"{Path(source_dir).name}_backup_{timestamp}"
    backup_path = Path(backup_dir) / backup_name
    
    # バックアップディレクトリを作成
    os.makedirs(backup_dir, exist_ok=True)
    
    # ディレクトリをコピー
    shutil.copytree(source_dir, backup_path)
    
    # ハッシュ値を計算して整合性を確認
    source_hash = calculate_directory_hash(source_dir)
    backup_hash = calculate_directory_hash(backup_path)
    
    backup_info = {
        'backup_path': str(backup_path),
        'source_path': source_dir,
        'timestamp': timestamp,
        'source_hash': source_hash,
        'backup_hash': backup_hash,
        'status': 'OK' if source_hash == backup_hash else 'WARNING'
    }
    
    # バックアップ情報を保存
    info_file = Path(backup_dir) / f"{backup_name}_info.json"
    write(file_path=str(info_file), content=json.dumps(backup_info, indent=2))
    
    return backup_info

def calculate_directory_hash(directory_path):
    """ディレクトリのハッシュ値を計算"""
    hash_md5 = hashlib.md5()
    
    for root, dirs, files in os.walk(directory_path):
        for file in files:
            file_path = os.path.join(root, file)
            with open(file_path, 'rb') as f:
                hash_md5.update(f.read())
    
    return hash_md5.hexdigest()

# 使用例
# ファイルを整理
organized_dir = organize_project_files("my_project")

# バックアップを作成
backup_info = create_backup("my_project")
print(f"バックアップ完了: {backup_info['backup_path']}")
```

### ワークフローの特徴
- **自動分類**: 拡張子からファイルを自動分類
- **バックアップ**: 整理前に自動バックアップ
- **整合性確認**: ハッシュ値でデータ整合性を確認

## ワークフロー4：メールと通知の自動処理

### �ケーススタディ
定期的なメール確認や通知処理を自動化します。特に重要な通知だけをピックアップして対応するワークフローは時間節約になります。

### 実装コード

```python
import smtplib
import imaplib
import email
from datetime import datetime, timedelta
import re

class NotificationProcessor:
    def __init__(self, email_config, priority_keywords=None):
        self.email_config = email_config
        self.priority_keywords = priority_keywords or ['緊急', '重要', 'error', 'critical']
        
    def process_new_notifications(self, days=1):
        """新しい通知を処理する"""
        # 指定日前からのメールを取得
        since_date = (datetime.now() - timedelta(days=days)).strftime("%d-%b-%Y")
        
        # メールを取得
        emails = self.fetch_emails(since_date)
        
        # 通知を処理
        processed_count = 0
        priority_count = 0
        
        for email_data in emails:
            if self.is_priority_notification(email_data['subject'], email_data['body']):
                self.handle_priority_notification(email_data)
                priority_count += 1
            
            self.handle_normal_notification(email_data)
            processed_count += 1
        
        # 処理レポートを生成
        self.generate_processing_report(processed_count, priority_count)
        
        return processed_count
    
    def fetch_emails(self, since_date):
        """IMAPでメールを取得"""
        try:
            # IMAP接続
            mail = imaplib.IMAP4_SSL(self.email_config['imap_server'])
            mail.login(self.email_config['email'], self.email_config['password'])
            mail.select('inbox')
            
            # メール検索
            status, messages = mail.search(None, f'(SINCE "{since_date}")')
            
            emails = []
            for message_id in messages[0].split():
                # メール取得
                status, msg_data = mail.fetch(message_id, '(RFC822)')
                
                # メール解析
                raw_email = msg_data[0][1]
                email_message = email.message_from_bytes(raw_email)
                
                # 件名と本文を抽出
                subject = email_message['Subject']
                body = self.extract_email_body(email_message)
                
                emails.append({
                    'id': message_id,
                    'subject': subject,
                    'body': body,
                    'from': email_message['From'],
                    'date': email_message['Date']
                })
            
            mail.logout()
            return emails
            
        except Exception as e:
            print(f"メール取得エラー: {e}")
            return []
    
    def extract_email_body(self, email_message):
        """メール本文を抽出"""
        body = ""
        
        if email_message.is_multipart():
            for part in email_message.walk():
                content_type = part.get_content_type()
                if content_type == "text/plain":
                    payload = part.get_payload(decode=True)
                    body += payload.decode('utf-8', errors='ignore')
        else:
            payload = email_message.get_payload(decode=True)
            body = payload.decode('utf-8', errors='ignore')
        
        return body
    
    def is_priority_notification(self, subject, body):
        """優先通知か判定"""
        text = f"{subject} {body}".lower()
        
        for keyword in self.priority_keywords:
            if keyword.lower() in text:
                return True
        
        return False
    
    def handle_priority_notification(self, email_data):
        """優先通知を処理"""
        # 優先通知をファイルに記録
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        priority_file = f"priority_notifications/通知_{timestamp}.md"
        
        content = f"""# 優先通知

## 件名
{email_data['subject']}

## 送信者
{email_data['from']}

## 日時
{email_data['date']}

## 内容
{email_data['body']}

---
処理日時: {datetime.now().strftime("%Y-%m-%d %H:%M:%S")}
"""
        
        write(file_path=priority_file, content=content)
        
        # 必要に応じてアラートを送信
        self.send_alert(email_data)
    
    def handle_normal_notification(self, email_data):
        """通常通知を処理"""
        # 通常通知をログに記録
        log_entry = f"[{email_data['date']}] {email_data['from']}: {email_data['subject']}"
        
        # ログファイルに追記
        log_file = "notification_log.txt"
        current_content = read(file_path=log_file) if os.path.exists(log_file) else ""
        
        new_content = f"{current_content}\n{log_entry}"
        write(file_path=log_file, content=new_content)
    
    def send_alert(self, email_data):
        """アラートを送信"""
        # 簡易なメールアラート
        alert_subject = f"【重要通知】{email_data['subject']}"
        alert_body = f"""
優先通知が検出されました。

件名: {email_data['subject']}
送信者: {email_data['from']}
日時: {email_data['date']}

内容:
{email_data['body']}
"""
        
        self.send_email(
            to=self.email_config['alert_email'],
            subject=alert_subject,
            body=alert_body
        )
    
    def send_email(self, to, subject, body):
        """メールを送信"""
        try:
            # SMTP接続
            server = smtplib.SMTP(self.email_config['smtp_server'], self.email_config['smtp_port'])
            server.starttls()
            server.login(self.email_config['email'], self.email_config['password'])
            
            # メール作成
            message = f"Subject: {subject}\nTo: {to}\n\n{body}"
            
            # 送信
            server.sendmail(self.email_config['email'], to, message)
            server.quit()
            
        except Exception as e:
            print(f"メール送信エラー: {e}")
    
    def generate_processing_report(self, processed_count, priority_count):
        """処理レポートを生成"""
        report_content = f"""# 通知処理レポート

## 処理概要
- 処理日時: {datetime.now().strftime("%Y-%m-%d %H:%M:%S")}
- 処理メール数: {processed_count}
- 優先通知数: {priority_count}

## 処理結果
- 通常通知: {processed_count - priority_count} 件
- 優先通知: {priority_count} 件
- 処理率: 100%

## 推奨アクション
- 優先通知を確認してください
- 通常通知は後で確認可能です
"""
        
        report_file = f"notification_reports/処理レポート_{datetime.now().strftime('%Y%m%d')}.md"
        write(file_path=report_file, content=report_content)

# 使用例
email_config = {
    'email': 'your-email@example.com',
    'password': 'your-password',
    'imap_server': 'imap.gmail.com',
    'smtp_server': 'smtp.gmail.com',
    'smtp_port': 587,
    'alert_email': 'alert@example.com'
}

processor = NotificationProcessor(email_config)
processed_count = processor.process_new_notifications(days=1)
print(f"処理完了: {processed_count} 件の通知を処理")
```

### ワークフローの特徴
- **自動フィルタリング**: 重要な通知を自動的に判別
- **優先度処理**: 優先通知を別途処理
- **処理ログ**: 全ての操作をログに記録

## ワークフロー5：コードレビュー準備の自動化

### ケーススタディ
コードレビューの準備は時間がかかります。変更点の抽出、ドキュメントの更新、テストの実行などを自動化します。

### 実装コード

```python
import subprocess
import re
from pathlib import Path

class CodeReviewPreparer:
    def __init__(self, repo_path):
        self.repo_path = repo_path
        self.changes = []
        
    def prepare_review_materials(self):
        """コードレビュー用の材料を準備"""
        print("コードレビュー準備を開始...")
        
        # 変更点を抽出
        self.extract_changes()
        
        # ドキュメントを更新
        self.update_documentation()
        
        # テストを実行
        test_results = self.run_tests()
        
        # レビューレポートを生成
        review_report = self.generate_review_report(test_results)
        
        # レビュー用ファイルを作成
        self.create_review_files(review_report)
        
        print("コードレビュー準備完了！")
        return review_report
    
    def extract_changes(self):
        """変更点を抽出"""
        try:
            # Gitのdiffを取得
            result = subprocess.run(
                ['git', 'diff', '--name-only'],
                cwd=self.repo_path,
                capture_output=True,
                text=True
            )
            
            if result.returncode == 0:
                changed_files = result.stdout.strip().split('\n')
                
                # 各ファイルの変更を詳細に分析
                for file_path in changed_files:
                    if file_path:  # 空行を除外
                        file_changes = self.analyze_file_changes(file_path)
                        self.changes.append(file_changes)
            
        except Exception as e:
            print(f"変更点抽出エラー: {e}")
    
    def analyze_file_changes(self, file_path):
        """ファイルの変更を分析"""
        try:
            # ファイルタイプ判定
            file_extension = Path(file_path).suffix.lower()
            
            # 変更行数を取得
            result = subprocess.run(
                ['git', 'diff', '--numstat', file_path],
                cwd=self.repo_path,
                capture_output=True,
                text=True
            )
            
            if result.returncode == 0:
                numstat = result.stdout.strip()
                if numstat:
                    added, deleted, _ = numstat.split('\t')
                    added_lines = int(added) if added.isdigit() else 0
                    deleted_lines = int(deleted) if deleted.isdigit() else 0
                else:
                    added_lines = deleted_lines = 0
            else:
                added_lines = deleted_lines = 0
            
            # 変更内容のサマリー
            change_summary = self.get_change_summary(file_path)
            
            return {
                'file_path': file_path,
                'file_type': file_extension,
                'added_lines': added_lines,
                'deleted_lines': deleted_lines,
                'total_changes': added_lines + deleted_lines,
                'change_summary': change_summary,
                'impact_level': self.assess_impact(added_lines, deleted_lines, file_extension)
            }
            
        except Exception as e:
            print(f"ファイル分析エラー {file_path}: {e}")
            return {
                'file_path': file_path,
                'error': str(e)
            }
    
    def get_change_summary(self, file_path):
        """変更のサマリーを取得"""
        try:
            result = subprocess.run(
                ['git', 'diff', '--unified=3', file_path],
                cwd=self.repo_path,
                capture_output=True,
                text=True
            )
            
            if result.returncode == 0:
                diff_content = result.stdout
                
                # 機能追加の検出
                feature_pattern = r'(?i)(add|feature|new).+function'
                features = re.findall(feature_pattern, diff_content)
                
                # バグ修正の検出
                bug_pattern = r'(?i)(fix|bug|error).+solution'
                bugs = re.findall(bug_pattern, diff_content)
                
                # リファクタリングの検出
                refactor_pattern = r'(?i)(refactor|optimize|improve).+code'
                refactors = re.findall(refactor_pattern, diff_content)
                
                return {
                    'features': len(features),
                    'bugs_fixed': len(bugs),
                    'refactors': len(refactors),
                    'summary': self.generate_change_summary(features, bugs, refactors)
                }
            
        except Exception as e:
            print(f"変更サマリー取得エラー: {e}")
        
        return {'summary': '変更の詳細を取得できませんでした'}
    
    def generate_change_summary(self, features, bugs, refactors):
        """変更サマリーを生成"""
        summary_parts = []
        
        if features:
            summary_parts.append(f"{len(features)}機能追加")
        if bugs:
            summary_parts.append(f"{len(bugs)}バグ修正")
        if refactors:
            summary_parts.append(f"{len(refactors)}リファクタリング")
        
        if summary_parts:
            return ", ".join(summary_parts)
        else:
            return "マイナーな変更"
    
    def assess_impact(self, added_lines, deleted_lines, file_extension):
        """変更の影響度を評価"""
        total_changes = added_lines + deleted_lines
        
        if total_changes == 0:
            return "none"
        elif total_changes < 10:
            return "low"
        elif total_changes < 50:
            return "medium"
        elif file_extension in ['.py', '.js', '.java', '.cpp']:
            return "high"
        else:
            return "medium"
    
    def update_documentation(self):
        """ドキュメントを更新"""
        try:
            # README.mdの更新日時を更新
            readme_path = Path(self.repo_path) / "README.md"
            if readme_path.exists():
                content = read(file_path=str(readme_path))
                
                # 更新日時を更新
                updated_content = re.sub(
                    r'最終更新:.+',
                    f'最終更新: {datetime.now().strftime("%Y-%m-%d %H:%M:%S")}',
                    content
                )
                
                write(file_path=str(readme_path), content=updated_content)
                
        except Exception as e:
            print(f"ドキュメント更新エラー: {e}")
    
    def run_tests(self):
        """テストを実行"""
        try:
            result = subprocess.run(
                ['python', '-m', 'pytest', '--tb=short'],
                cwd=self.repo_path,
                capture_output=True,
                text=True
            )
            
            return {
                'exit_code': result.returncode,
                'stdout': result.stdout,
                'stderr': result.stderr,
                'tests_passed': 'passed' in result.stdout.lower(),
                'test_summary': self.parse_test_output(result.stdout)
            }
            
        except Exception as e:
            return {
                'error': str(e),
                'tests_passed': False
            }
    
    def parse_test_output(self, output):
        """テスト出力を解析"""
        try:
            # テスト結果のパターンマッチング
            passed_pattern = r'(\d+) passed'
            failed_pattern = r'(\d+) failed'
            
            passed = re.search(passed_pattern, output)
            failed = re.search(failed_pattern, output)
            
            return {
                'passed': int(passed.group(1)) if passed else 0,
                'failed': int(failed.group(1)) if failed else 0
            }
            
        except Exception:
            return {'passed': 0, 'failed': 0}
    
    def generate_review_report(self, test_results):
        """レビューレポートを生成"""
        total_changed_files = len([c for c in self.changes if 'error' not in c])
        total_added_lines = sum(c.get('added_lines', 0) for c in self.changes if 'error' not in c)
        total_deleted_lines = sum(c.get('deleted_lines', 0) for c in self.changes if 'error' not in c)
        
        # 影響度レベルの集計
        impact_counts = {}
        for change in self.changes:
            if 'error' not in change:
                impact = change.get('impact_level', 'low')
                impact_counts[impact] = impact_counts.get(impact, 0) + 1
        
        report_content = f"""# コードレビューレポート

## 変更概要
- 変更ファイル数: {total_changed_files}
- 追加行数: {total_added_lines}
- 削除行数: {total_deleted_lines}
- 総変更行数: {total_added_lines + total_deleted_lines}

## テスト結果
"""
        
        if test_results.get('tests_passed', False):
            report_content += "✅ テストはすべてパスしました\n"
        else:
            report_content += "❌ テストに失敗しています\n"
        
        if 'test_summary' in test_results:
            summary = test_results['test_summary']
            report_content += f"パス: {summary.get('passed', 0)}, 失敗: {summary.get('failed', 0)}\n"
        
        report_content += f"""

## 影響度分布
"""
        for impact, count in impact_counts.items():
            impact_emoji = {'high': '🔴', 'medium': '🟡', 'low': '🟢', 'none': '⚪'}.get(impact, '⚪')
            report_content += f"- {impact_emoji} {impact}: {count} ファイル\n"
        
        report_content += "\n## 変更ファイル詳細\n"
        
        for change in self.changes:
            if 'error' not in change:
                report_content += f"\n### {change['file_path']}\n"
                report_content += f"- ファイルタイプ: {change['file_type']}\n"
                report_content += f"- 追加行: {change['added_lines']}, 削除行: {change['deleted_lines']}\n"
                report_content += f"- 影響度: {change['impact_level']}\n"
                report_content += f"- 変更内容: {change['change_summary']['summary']}\n"
        
        return report_content
    
    def create_review_files(self, review_report):
        """レビュー用ファイルを作成"""
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        
        # レビューレポート
        report_file = f"code_review/review_report_{timestamp}.md"
        write(file_path=report_file, content=review_report)
        
        # 変更ファイルリスト

        # レビューチェックリスト

        return {
            'report_file': report_file,
            'changes_file': changes_file,
            'checklist_file': checklist_file
        }
    
    def generate_review_checklist(self):
        """レビューチェックリストを生成"""
        checklist = """# コードレビューチェックリスト

## 基本的な確認項目

---
チェックリスト作成日時: {timestamp}

        return checklist.format(timestamp=datetime.now().strftime("%Y-%m-%d %H:%M:%S"))
```

## ワークフロー6：定期的なデータ同期とバックアップ

### ケーススタディ
複数の環境（開発・ staging・本番）間でデータを同期したり、定期的にバックアップを取得したりする必要がある場合。

### 実装コード

```python
import shutil
import time
from datetime import datetime, timedelta
import hashlib
import json

class DataSyncManager:
    def __init__(self, config_file="sync_config.json"):
        self.config = self.load_config(config_file)
        self.sync_log = []
        
    def load_config(self, config_file):
        """設定ファイルを読み込む"""
        try:
            content = read(file_path=config_file)
            return json.loads(content)
        except Exception as e:
            print(f"設定ファイル読み込みエラー: {e}")
            return self.create_default_config()
    
    def create_default_config(self):
        """デフォルト設定を作成"""
        default_config = {
            "environments": {
                "development": {
                    "path": "./dev_data",
                    "type": "local"
                },
                "staging": {
                    "path": "./staging_data", 
                    "type": "local"
                },
                "production": {
                    "path": "./prod_data",
                    "type": "local"
                }
            },
            "sync_rules": [
                {
                    "source": "development",
                    "target": "staging",
                    "schedule": "daily",
                    "sync_type": "incremental"
                },
                {
                    "source": "staging",
                    "target": "production", 
                    "schedule": "weekly",
                    "sync_type": "full"
                }
            ],
            "backup_rules": [
                {
                    "source": "production",
                    "schedule": "daily",
                    "retention_days": 7
                }
            ]
        }
        
        write(file_path="sync_config.json", content=json.dumps(default_config, indent=2))
        return default_config
    
    def run_sync_cycle(self):
        """同期サイクルを実行"""
        print("データ同期サイクルを開始...")
        
        # 同期を実行
        sync_results = []
        for rule in self.config["sync_rules"]:
            result = self.execute_sync_rule(rule)
            sync_results.append(result)
        
        # バックアップを実行
        backup_results = []
        for rule in self.config["backup_rules"]:
            result = self.execute_backup_rule(rule)
            backup_results.append(result)
        
        # サイクルレポートを生成
        cycle_report = self.generate_cycle_report(sync_results, backup_results)
        
        print("データ同期サイクル完了！")
        return cycle_report
    
    def execute_sync_rule(self, rule):
        """同期ルールを実行"""
        source_env = self.config["environments"][rule["source"]]
        target_env = self.config["environments"][rule["target"]]
        
        print(f"同期実行: {rule['source']} -> {rule['target']}")
        
        try:
            if rule["sync_type"] == "full":
                # 完全同期
                result = self.full_sync(source_env, target_env)
            else:
                # 増分同期
                result = self.incremental_sync(source_env, target_env)
            
            # 同期結果をログに記録
            self.log_sync(rule, result)
            
            return {
                "rule": rule,
                "success": True,
                "details": result
            }
            
        except Exception as e:
            print(f"同期エラー: {e}")
            return {
                "rule": rule,
                "success": False,
                "error": str(e)
            }
    
    def full_sync(self, source_env, target_env):
        """完全同期を実行"""
        # ターゲットディレクトリをバックアップ
        backup_path = self.backup_directory(target_env["path"])
        
        # ソースディレクトリをコピー
        sync_start = time.time()
        shutil.copytree(source_env["path"], target_env["path"])
        sync_time = time.time() - sync_start
        
        # 同期情報を収集
        file_count = self.count_files(source_env["path"])
        
        return {
            "type": "full",
            "sync_time": sync_time,
            "file_count": file_count,
            "backup_path": backup_path
        }
    
    def incremental_sync(self, source_env, target_env):
        """増分同期を実行"""
        # 変更ファイルを検出
        changed_files = self.detect_changed_files(source_env["path"], target_env["path"])
        
        # 変更ファイルを同期
        sync_count = 0
        for file_path in changed_files:
            source_file = os.path.join(source_env["path"], file_path)
            target_file = os.path.join(target_env["path"], file_path)
            
            # ディレクトリを作成
            os.makedirs(os.path.dirname(target_file), exist_ok=True)
            
            # ファイルをコピー
            shutil.copy2(source_file, target_file)
            sync_count += 1
        
        return {
            "type": "incremental",
            "synced_files": sync_count,
            "changed_files": len(changed_files)
        }
    
    def detect_changed_files(self, source_path, target_path):
        """変更されたファイルを検出"""
        changed_files = []
        
        for root, dirs, files in os.walk(source_path):
            for file in files:
                source_file = os.path.join(root, file)
                rel_path = os.path.relpath(source_file, source_path)
                target_file = os.path.join(target_path, rel_path)
                
                # ターゲットファイルが存在しない、または内容が異なる場合
                if not os.path.exists(target_file):
                    changed_files.append(rel_path)
                else:
                    source_hash = self.calculate_file_hash(source_file)
                    target_hash = self.calculate_file_hash(target_file)
                    
                    if source_hash != target_hash:
                        changed_files.append(rel_path)
        
        return changed_files
    
    def calculate_file_hash(self, file_path):
        """ファイルのハッシュ値を計算"""
        hash_md5 = hashlib.md5()
        with open(file_path, 'rb') as f:
            hash_md5.update(f.read())
        return hash_md5.hexdigest()
    
    def backup_directory(self, dir_path):
        """ディレクトリをバックアップ"""
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        backup_name = f"backup_{timestamp}"
        backup_path = os.path.join("backups", backup_name)
        
        # バックアップディレクトリを作成
        os.makedirs(os.path.dirname(backup_path), exist_ok=True)
        
        # ディレクトリをコピー
        shutil.copytree(dir_path, backup_path)
        
        # バックアップ情報を保存
        backup_info = {
            "original_path": dir_path,
            "backup_path": backup_path,
            "timestamp": timestamp,
            "file_count": self.count_files(dir_path)
        }
        
        info_file = f"backups/{backup_name}_info.json"
        write(file_path=info_file, content=json.dumps(backup_info, indent=2))
        
        return backup_path
    
    def execute_backup_rule(self, rule):
        """バックアップルールを実行"""
        source_env = self.config["environments"][rule["source"]]
        
        print(f"バックアップ実行: {rule['source']}")
        
        try:
            backup_path = self.backup_directory(source_env["path"])
            
            # 古いバックアップを削除
            self.cleanup_old_backups(rule["retention_days"])
            
            return {
                "source": rule["source"],
                "success": True,
                "backup_path": backup_path
            }
            
        except Exception as e:
            print(f"バックアップエラー: {e}")
            return {
                "source": rule["source"],
                "success": False,
                "error": str(e)
            }
    
    def cleanup_old_backups(self, retention_days):
        """古いバックアップを削除"""
        cutoff_time = datetime.now() - timedelta(days=retention_days)
        
        backups_dir = Path("backups")
        if backups_dir.exists():
            for backup_dir in backups_dir.iterdir():
                if backup_dir.is_dir() and backup_dir.name.startswith("backup_"):
                    try:
                        # バックアップ情報ファイルからタイムスタンプ取得
                        info_file = backup_dir / f"{backup_dir.name}_info.json"
                        if info_file.exists():
                            content = read(file_path=str(info_file))
                            info = json.loads(content)
                            backup_time = datetime.strptime(info["timestamp"], "%Y%m%d_%H%M%S")
                            
                            if backup_time < cutoff_time:
                                shutil.rmtree(backup_dir)
                                print(f"古いバックアップを削除: {backup_dir}")
                    
                    except Exception as e:
                        print(f"バックアップ削除エラー: {e}")
    
    def log_sync(self, rule, result):
        """同期ログを記録"""
        log_entry = {
            "timestamp": datetime.now().isoformat(),
            "rule": rule,
            "result": result
        }
        
        self.sync_log.append(log_entry)
        
        # ログファイルに保存
        log_file = "sync_log.json"
        current_logs = json.loads(read(file_path=log_file)) if os.path.exists(log_file) else []
        current_logs.append(log_entry)
        
        write(file_path=log_file, content=json.dumps(current_logs, indent=2))
    
    def generate_cycle_report(self, sync_results, backup_results):
        """サイクルレポートを生成"""
        total_syncs = len(sync_results)
        successful_syncs = sum(1 for r in sync_results if r["success"])
        
        total_backups = len(backup_results)
        successful_backups = sum(1 for r in backup_results if r["success"])
        
        report_content = f"""# データ同期サイクルレポート

## サイクル概要
- 実行日時: {datetime.now().strftime("%Y-%m-%d %H:%M:%S")}
- 同期ルール数: {total_syncs}
- バックアップルール数: {total_backups}

## 同期結果
- 成功: {successful_syncs}/{total_syncs}
- 失敗: {total_syncs - successful_syncs}/{total_syncs}

## バックアップ結果
- 成功: {successful_backups}/{total_backups}
- 失敗: {total_backups - successful_backups}/{total_backups}

## 詳細結果
"""
        
        for result in sync_results:
            status = "✅ 成功" if result["success"] else "❌ 失敗"
            report_content += f"\n### 同期: {result['rule']['source']} -> {result['rule']['target']}\n"
            report_content += f"- 状態: {status}\n"
            
            if result["success"]:
                details = result["details"]
                if details["type"] == "full":
                    report_content += f"- タイプ: 完全同期\n"
                    report_content += f"- 同期ファイル数: {details['file_count']}\n"
                else:
                    report_content += f"- タイプ: 増分同期\n"
                    report_content += f"- 変更ファイル数: {details['changed_files']}\n"
                    report_content += f"- 同期ファイル数: {details['synced_files']}\n"
            else:
                report_content += f"- エラー: {result['error']}\n"
        
        for result in backup_results:
            status = "✅ 成功" if result["success"] else "❌ 失敗"
            report_content += f"\n### バックアップ: {result['source']}\n"
            report_content += f"- 状態: {status}\n"
            
            if result["success"]:
                report_content += f"- バックアップ先: {result['backup_path']}\n"
            else:
                report_content += f"- エラー: {result['error']}\n"
        
        # レポートを保存
        timestamp = datetime.now().strftime("%Y%m%d_%H%M%S")
        report_file = f"sync_reports/cycle_report_{timestamp}.md"
        write(file_path=report_file, content=report_content)
        
        return {
            "report_file": report_file,
            "sync_success_rate": successful_syncs / total_syncs if total_syncs > 0 else 1,
            "backup_success_rate": successful_backups / total_backups if total_backups > 0 else 1,
            "total_operations": total_syncs + total_backups,
            "successful_operations": successful_syncs + successful_backups
        }
    
    def count_files(self, directory):
        """ディレクトリ内のファイル数をカウント"""
        count = 0
        for root, dirs, files in os.walk(directory):
            count += len(files)
        return count

# 使用例
sync_manager = DataSyncManager()
cycle_report = sync_manager.run_sync_cycle()
```

## ワークフロー7：リマインダーとタスク管理

### ケーススタディ
定期的なタスクや期限のある作業を忘れずに管理する必要があります。OpenClawを使ってリマインダーとタスク管理を自動化します。

### 実装コード

```python
import json
from datetime import datetime, timedelta
import time
import threading

class TaskManager:
    def __init__(self, task_file="tasks.json", reminder_file="reminders.json"):
        self.task_file = task_file
        self.reminder_file = reminder_file
        self.tasks = self.load_tasks()
        self.reminders = self.load_reminders()
        self.running = False
        
    def load_tasks(self):
        """タスクを読み込む"""
        try:
            content = read(file_path=self.task_file)
            return json.loads(content)
        except:
            return []
    
    def load_reminders(self):
        """リマインダーを読み込む"""
        try:
            content = read(file_path=self.reminder_file)
            return json.loads(content)
        except:
            return []
    
    def save_tasks(self):
        """タスクを保存"""
        write(file_path=self.task_file, content=json.dumps(self.tasks, indent=2))
    
    def save_reminders(self):
        """リマインダーを保存"""
        write(file_path=self.reminder_file, content=json.dumps(self.reminders, indent=2))
    
    def add_task(self, title, description="", due_date=None, priority="medium"):
        """タスクを追加"""
        task = {
            "id": len(self.tasks) + 1,
            "title": title,
            "description": description,
            "due_date": due_date,
            "priority": priority,
            "created_at": datetime.now().isoformat(),
            "status": "pending",
            "completed_at": None
        }
        
        self.tasks.append(task)
        self.save_tasks()
        
        # 期限に合わせてリマインダーを作成
        if due_date:
            self.create_reminder(task)
        
        return task["id"]
    
    def create_reminder(self, task):
        """タスク用のリマインダーを作成"""
        due_date = datetime.fromisoformat(task["due_date"])
        
        # 1日前、3日前、1週間前のリマインダー
        reminder_times = [
            due_date - timedelta(days=1),
            due_date - timedelta(days=3),
            due_date - timedelta(weeks=1)
        ]
        
        for reminder_time in reminder_times:
            reminder = {
                "id": len(self.reminders) + 1,
                "task_id": task["id"],
                "title": f"期限リマインダー: {task['title']}",
                "message": self.generate_reminder_message(task, reminder_time),
                "reminder_time": reminder_time.isoformat(),
                "triggered": False,
                "created_at": datetime.now().isoformat()
            }
            
            self.reminders.append(reminder)
        
        self.save_reminders()
    
    def generate_reminder_message(self, task, reminder_time):
        """リマインダーメッセージを生成"""
        due_date = datetime.fromisoformat(task["due_date"])
        days_until_due = (due_date - reminder_time).days
        
        if days_until_due == 1:
            time_text = "明日"
        elif days_until_due == 3:
            time_text = "3日後"
        elif days_until_due == 7:
            time_text = "1週間後"
        else:
            time_text = f"{days_until_due}日後"
        
        message = f"""
【{task['priority'].upper()}】タスクの期限が{time_text}です！

タスク名: {task['title']}
期限: {task['due_date']}
説明: {task['description']}

期限内に完了してください。
"""
        
        return message.strip()
    
    def complete_task(self, task_id):
        """タスクを完了"""
        for task in self.tasks:
            if task["id"] == task_id:
                task["status"] = "completed"
                task["completed_at"] = datetime.now().isoformat()
                
                # 関連するリマインダーを無効化
                for reminder in self.reminders:
                    if reminder["task_id"] == task_id:
                        reminder["triggered"] = True
                
                self.save_tasks()
                self.save_reminders()
                return True
        
        return False
    
    def start_monitoring(self):
        """モニタリングを開始"""
        self.running = True
        monitor_thread = threading.Thread(target=self.monitor_loop)
        monitor_thread.daemon = True
        monitor_thread.start()
        print("タスクモニタリングを開始しました")
    
    def monitor_loop(self):
        """モニタリングループ"""
        while self.running:
            current_time = datetime.now()
            
            # リマインダーをチェック
            for reminder in self.reminders:
                if not reminder["triggered"]:
                    reminder_time = datetime.fromisoformat(reminder["reminder_time"])
                    
                    if current_time >= reminder_time:
                        self.trigger_reminder(reminder)
                        reminder["triggered"] = True
            
            # 期限切れタスクをチェック
            self.check_overdue_tasks()
            
            # 1分ごとにチェック
            time.sleep(60)
    
    def trigger_reminder(self, reminder):
        """リマインダーをトリガー"""
        print(f"\n🔔 リマインダー:")
        print(reminder["message"])
        
        # リマインダーファイルに記録
        timestamp = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        log_entry = f"[{timestamp}] {reminder['title']}\n{reminder['message']}\n"
        
        log_file = "reminder_log.txt"
        current_content = read(file_path=log_file) if os.path.exists(log_file) else ""
        new_content = f"{current_content}\n{log_entry}"
        
        write(file_path=log_file, content=new_content)
    
    def check_overdue_tasks(self):
        """期限切れタスクをチェック"""
        current_time = datetime.now()
        overdue_tasks = []
        
        for task in self.tasks:
            if task["status"] == "pending" and task["due_date"]:
                due_date = datetime.fromisoformat(task["due_date"])
                
                if current_time > due_date:
                    overdue_tasks.append(task)
        
        if overdue_tasks:
            print(f"\n⚠️ 期限切れタスク ({len(overdue_tasks)}件):")
            for task in overdue_tasks:
                print(f"- {task['title']} (期限: {task['due_date']})")
    
    def get_daily_summary(self):
        """日次サマリーを生成"""
        today = datetime.now().strftime("%Y-%m-%d")
        
        # 今日のタスク
        today_tasks = []
        overdue_tasks = []
        
        for task in self.tasks:
            if task["due_date"] and task["due_date"].startswith(today):
                today_tasks.append(task)
            elif task["status"] == "pending" and task["due_date"]:
                due_date = datetime.fromisoformat(task["due_date"])
                if datetime.now() > due_date:
                    overdue_tasks.append(task)
        
        # 完成したタスク
        completed_tasks = [t for t in self.tasks if t["status"] == "completed"]
        
        summary_content = f"""# 日次タスクリマインダー

## 今日の日付
{today}

## 今日の期限タスク ({len(today_tasks)}件)
"""
        
        for task in today_tasks:
            priority_icon = {"high": "🔴", "medium": "🟡", "low": "🟢"}.get(task["priority"], "⚪")
            summary_content += f"- {priority_icon} {task['title']}\n"
        
        if overdue_tasks:
            summary_content += f"\n## 期限切れタスク ({len(overdue_tasks)}件)\n"
            for task in overdue_tasks:
                summary_content += f"- ⚠️ {task['title']} (期限: {task['due_date']})\n"
        
        summary_content += f"\n## 完成タスク ({len(completed_tasks)}件)\n"
        
        # 最近完成したタスクを5件表示
        recent_completed = sorted(
            [t for t in completed_tasks if t["completed_at"]],
            key=lambda x: x["completed_at"],
            reverse=True
        )[:5]
        
        for task in recent_completed:
            completed_time = datetime.fromisoformat(task["completed_at"]).strftime("%H:%M")
            summary_content += f"- ✅ {task['title']} ({completed_time})\n"
        
        # サマリーを保存
        summary_file = f"daily_summaries/summary_{today}.md"
        write(file_path=summary_file, content=summary_content)
        
        return {
            "today_tasks": len(today_tasks),
            "overdue_tasks": len(overdue_tasks),
            "completed_tasks": len(completed_tasks),
            "summary_file": summary_file
        }
    
    def generate_weekly_report(self):
        """週次レポートを生成"""
        # 1週間前の日付を計算
        week_ago = datetime.now() - timedelta(weeks=1)
        
        # 週間データを収集
        week_completed = [t for t in self.tasks 
                          if t["status"] == "completed" and 
                          t["completed_at"] and 
                          datetime.fromisoformat(t["completed_at"]) >= week_ago]
        
        week_created = [t for t in self.tasks 
                      if datetime.fromisoformat(t["created_at"]) >= week_ago]
        
        # 優先度別集計
        priority_stats = {"high": 0, "medium": 0, "low": 0}
        for task in week_completed:
            priority_stats[task["priority"]] += 1
        
        report_content = f"""# 週次タスクリポート

## 期間
{week_ago.strftime('%Y-%m-%d')} - {datetime.now().strftime('%Y-%m-%d')}

## 集計結果
- 作成タスク: {len(week_created)} 件
- 完成タスク: {len(week_completed)} 件
- 完成率: {len(week_completed) / max(len(week_created), 1) * 100:.1f}%

## 優先度別内訳
- 高: {priority_stats['high']} 件
- 中: {priority_stats['medium']} 件
- 低: {priority_stats['low']} 件

## 完成したタスク
"""
        
        # 最近完成したタスクを表示
        recent_completed = sorted(
            week_completed,
            key=lambda x: x["completed_at"],
            reverse=True
        )
        
        for task in recent_completed:
            completed_date = datetime.fromisoformat(task["completed_at"]).strftime("%m/%d %H:%M")
            priority_icon = {"high": "🔴", "medium": "🟡", "low": "🟢"}.get(task["priority"], "⚪")
            report_content += f"- {priority_icon} {task['title']} ({completed_date})\n"
        
        # レポートを保存
        report_file = f"weekly_reports/week_{datetime.now().strftime('%Y%m%d')}.md"
        write(file_path=report_file, content=report_content)
        
        return {
            "completed_count": len(week_completed),
            "created_count": len(week_created),
            "completion_rate": len(week_completed) / max(len(week_created), 1),
            "report_file": report_file
        }

# 使用例
task_manager = TaskManager()

# タスクを追加
task1_id = task_manager.add_task(
    title="プロジェクト提案書の作成",
    description="クライアント向けの提案書を作成する",
    due_date=(datetime.now() + timedelta(days=3)).isoformat(),
    priority="high"
)

task2_id = task_manager.add_task(
    title="チームミーティングの準備",
    description="週次チームミーティングの議題準備",
    due_date=(datetime.now() + timedelta(days=1)).isoformat(),
    priority="medium"
)

# モニタリングを開始
task_manager.start_monitoring()

# 日次サマリーを取得
daily_summary = task_manager.get_daily_summary()
print(f"日次サマリー: {daily_summary['summary_file']}")

# 週次レポートを生成
weekly_report = task_manager.generate_weekly_report()
print(f"週次レポート: {weekly_report['report_file']}")
```

### ワークフローの特徴
- **自動リマインダー**: 期限に合わせて自動で通知
- **モニタリング**: 定期的なタスク状態の監視
- **レポート生成**: 日次・週次の自動レポート生成

## まとめ：7つの実用的なワークフロー

この記事では、OpenClawを使った7つの実用的なワークフローを紹介しました。各ワークフローは、以下のような日常業務の課題を解決します：

### 1. 進捗レポート自動生成
- **課題**: 手動での進捗報告が面倒
- **解決**: 自動情報収集とレポート生成

### 2. データ変換バッチ処理
- **課題**: 複数ファイルの形式変換が大変
- **解決**: 一括変換とエラーハンドリング

### 3. ファイル整理とバックアップ
- **課題**: ファイルが散在して管理が困難
- **解決**: 自動整理と安全なバックアップ

### 4. メールと通知処理
- **課題**: 重要な通知を見逃す
- **解決**: 自動フィルタリングと優先度処理

### 5. コードレビュー準備
- **課題**: レビュー準備に時間がかかる
- **解決**: 変更点分析とレビューレポート生成

### 6. データ同期とバックアップ
- **課題**: 複数環境間の同期が手間
- **解決**: 自動同期とバージョン管理

### 7. リマインダーとタスク管理
- **課題**: 期限を忘れてしまう
- **解決**: 自動リマインダーと進捗管理
